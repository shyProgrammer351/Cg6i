# <>  
	  
# TRUE   1  
# FALSE  0  
	  
# SCISSOR 1  
# ROCK    2  
# PAPER   3  
	  
# DoesPlayerAWin( playerA, playerB )  
	( ( ( playerA  SCISSOR )  ( playerB  PAPER ) )  
		 ( ( playerA  ROCK )  ( playerB  SCISSOR ) )  
		 ( ( playerA  PAPER )  ( playerB  ROCK ) ) )  
	   TRUE  FALSE  
	  
 getRockPaperScissors( char name[4]) {  
	printf( "%s, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: ", name );  
	 player = 0;  
	scanf( "%d",&player );  
	  
	( player ) {  
	 SCISSOR    
		printf( "가위를 냈습니다.\n" );  
		;  
	 ROCK    
		printf( "바위를 냈습니다.\n" );  
		;  
	 PAPER    
		printf( "보를 냈습니다.\n" );  
	}  
	  
	 player;  
}  
	  
int main() { 
	 playerA =  
	              getRockPaperScissors( "A양" );  
	 playerB =  
	              getRockPaperScissors( "B군" );  
	  
	 ( playerA  playerB )  
		printf( "A양과 B군이 비겼습니다." );  
	 ( DoesPlayerAWin( playerA, playerB )  
	                                              TRUE )  
		printf( "A양이 이겼습니다." );  
	  
		printf( "B군이 이겼습니다." );  
	  
	return 0; 
}
