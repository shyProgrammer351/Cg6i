	#include <stdio.h>
		
	#define TRUE   1
	#define FALSE  0
		
	#define SCISSOR 1
	#define ROCK    2
	#define PAPER   3
		
F1	#define DoesPlayerAWin( playerA, playerB )
		( ( ( playerA == SCISSOR ) && ( playerB == PAPER ) )
			|| ( ( playerA == ROCK ) && ( playerB == SCISSOR ) )
			|| ( ( playerA == PAPER ) && ( playerB == ROCK ) ) )
		? TRUE : FALSE
		
F2b	int getRockPaperScissors( char name[4] ) {
		printf( "%s, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: ", name );
		int player = 0;
		scanf( "%d", &player );
		
		switch( player ) {
		case SCISSOR:
			printf( "가위를 냈습니다.\n" );
			break;
		case ROCK:
			printf( "바위를 냈습니다.\n" );
			break;
		case PAPER:
			printf( "보를 냈습니다.\n" );
		}
		
F21		return player;
