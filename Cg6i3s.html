문제

A양과 B군이 가위바위보를 선택하면 승패를 출력하는 문제입니다 이를 해결하는 다음 프로그램의 실행순서를 클릭하세요 
A양, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: 2
바위를 냈습니다.
B군, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: 1
가위를 냈습니다.
A양이 이겼습니다.

프로그램 코드

	#include <stdio.h>

		

	#define TRUE   1

	#define FALSE  0

		

	#define SCISSOR 1

	#define ROCK    2

	#define PAPER   3

		

F1	#define DoesPlayerAWin( playerA, playerB )

		( ( ( playerA == SCISSOR ) && ( playerB == PAPER ) )

			|| ( ( playerA == ROCK ) && ( playerB == SCISSOR ) )

			|| ( ( playerA == PAPER ) && ( playerB == ROCK ) ) )

		? TRUE : FALSE

		

F2b	int getRockPaperScissors( char name[4] ) {

		printf( "%s, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: ", name );

		int player = 0;

		scanf( "%d", &player );

		

		switch( player ) {

		case SCISSOR:

			printf( "가위를 냈습니다.\n" );

			break;

		case ROCK:

			printf( "바위를 냈습니다.\n" );

			break;

		case PAPER:

			printf( "보를 냈습니다.\n" );

		}

		

F21		return player;

F2e	}

		

1	int main() {

2		int playerA =

3		              getRockPaperScissors( "A양" );

4		int playerB =

5		              getRockPaperScissors( "B군" );

		

6		if ( playerA == playerB )

7			printf( "A양과 B군이 비겼습니다." );

8		else if ( DoesPlayerAWin( playerA, playerB )

9		                                             == TRUE )

10			printf( "A양이 이겼습니다." );

11		else

12			printf( "B군이 이겼습니다." );

		

13		return 0;

	}
