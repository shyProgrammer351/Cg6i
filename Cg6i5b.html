Player move( char direction, Player player, int stage ) {  
	 ( ( ( direction  'w' ) || ( direction  'W' ) ) && ( player.row > 0 ) && ( map[stage].tile[player.row-1 ][player.column] WALL ) )  
		player.row = player.row - 1;  
	 ( ( ( direction  'a' ) || ( direction  'A' ) ) && ( player.column > 0 ) && ( map[stage].tile[player.row ][player.column-1] WALL ) )  
		player.column = player.column - 1;  
	 ( ( ( direction  's' ) || ( direction  'S' ) ) && ( player.row < (map[stage].height-1) ) && ( map[stage].tile[player.row+1 ][player.column] WALL ) )  
		player.row = player.row + 1;  
	 ( ( ( direction  'd' ) || ( direction  'D' ) ) && ( player.column < (map[stage].width-1) ) && ( map[stage].tile[player.row ][player.column+1] WALL ) )  
		player.column = player.column + 1;  
	  
	 player;  
}  
	  
Player updateStat( Player player, int stage ) {  
	 ( EXIT < map[stage].tile[player.row ][player.column]) {  
		player.hp += map[stage].tile[player.row ][player.column];  
		map[stage].tile[player.row ][player.column]= 0;  
	}  
	  
	 player;  
}

 showMap( int stage ) {  
	char symbol[10 ][64]= { "  ", " \033[44m   \033[0m", "옷", " \033[34m문 \033[0m", " \033[31m♥  \033[0m", " \033[33m★  \033[0m", " \033[32m♣  \033[0m", " \033[31m♠  \033[0m", " \033[36m◆  \033[0m", " \033[35m■  \033[0m" };  
	printf("\033[2J\033[1;1fHP 0\n");  
	 ( int row = 0; row < map[stage].height; row ++ ) {  
		 ( int column = 0; column < map[stage].width; column ++ )  
			printf( "%s", symbol[map[stage].tile[row ][column]]  );  
		printf( "\n" );  
	}  
}
